cmake_minimum_required(VERSION 3.15)

# CMP0091: select the MSVC runtime through the CMAKE_MSVC_RUNTIME_LIBRARY
# abstraction instead of hard-coded /MD flags, so the static-CRT (/MT) request
# in the MSVC branch below is actually honored. Defaults to NEW at >= 3.15, set
# explicitly so older generators behave deterministically too.
if(POLICY CMP0091)
    cmake_policy(SET CMP0091 NEW)
endif()

project(CAELUS_OS LANGUAGES CXX)

# ─── Language & Standard ──────────────────────────────────────────────────────
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ─── Build Options ────────────────────────────────────────────────────────────
option(CAELUS_EMBEDDED_UI
    "Embed UI assets from include/ui_payload.h (generated by build.bat / build.sh)"
    ON)
option(CAELUS_STATIC_BINARY
    "Link everything statically to produce a single portable executable"
    ON)

# ─── Compiler & Linker Flags (Phase 3: size-first, single static binary) ──────
# Compiler-conditional matrix — MSVC (cl.exe) vs GCC/Clang/MinGW. Both legs target
# a self-contained executable under the <50 MB budget, so flags favor SIZE +
# static linkage. The C++ standard (/std:c++17 resp. -std=c++17) is supplied by
# CMAKE_CXX_STANDARD above for both toolchains.
if(MSVC)
    # Static CRT (/MT) → no VCRUNTIME140.dll / msvcp140.dll dependency, keeping a
    # single portable file. Only takes effect with CMP0091 NEW (set at the top).
    # NOTE: a fully clean /MT link expects the Rust MSVC staticlib to also be built
    # with crt-static (RUSTFLAGS="-C target-feature=+crt-static"); the Rust side is
    # out of this task's two-file scope — otherwise the linker reconciles the CRTs
    # (LNK4098). See the integration note in the build report.
    if(CAELUS_STATIC_BINARY)
        set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
    endif()
    # /O2 = speed (swap to /O1 for pure size), /GL = whole-program optimization
    # (LTO equivalent), /EHsc = standard C++ exception model, /DNDEBUG = drop
    # asserts in the release image.
    # NOTE: /GS (stack buffer-overflow canaries) is LEFT ON deliberately — the old
    # "/GS-" disabled stack canaries on a binary that parses untrusted input.
    add_compile_options(/O2 /GL /EHsc /DNDEBUG)
    # /LTCG pairs with /GL; /INCREMENTAL:NO + /OPT:REF + /OPT:ICF shrink the image
    # (drop unreferenced funcs/data and fold identical COMDATs).
    add_link_options(/LTCG /INCREMENTAL:NO /OPT:REF /OPT:ICF)
else()
    # GCC / Clang / MinGW. -Os optimizes for SIZE (honors the <50 MB budget), -flto
    # adds link-time optimization for further shrinkage. Exceptions are KEPT enabled
    # (the C++ uses std::make_unique, std::string, <stdexcept>; "-fno-exceptions"
    # would turn any throw into std::terminate — a denial of service).
    add_compile_options(-Os -flto)
    add_link_options(-flto)
    if(CAELUS_STATIC_BINARY AND NOT APPLE)
        # Full static linkage (Windows/MinGW + Linux) and strip symbols at link time
        # (-s) for a small, reverse-engineering-resistant single binary.
        # NOTE: no "-Wl,--allow-multiple-definition" — when the Rust staticlib ABI
        # matches the C++ toolchain there are no duplicate symbols to suppress.
        add_link_options(-static -s)
    elseif(APPLE)
        # macOS: no -static for system libs and ld warns on -s; strip is handled by
        # the post-build custom command below instead.
        add_link_options(-static-libgcc -static-libstdc++)
    endif()
endif()

# ─── Source Files ─────────────────────────────────────────────────────────────
set(CAELUS_SOURCES
    "${CMAKE_CURRENT_SOURCE_DIR}/core_engine.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/src/intel_core.cpp"
)

add_executable(caelus_os ${CAELUS_SOURCES})

# ─── Include Paths ────────────────────────────────────────────────────────────
target_include_directories(caelus_os PRIVATE
    "${CMAKE_CURRENT_SOURCE_DIR}/include"   # ui_payload.h lives here
    "${CMAKE_CURRENT_SOURCE_DIR}/src"       # intel_core.h lives here
)

# ─── Embedded UI Feature Flag ─────────────────────────────────────────────────
if(CAELUS_EMBEDDED_UI)
    set(PAYLOAD_H "${CMAKE_CURRENT_SOURCE_DIR}/include/ui_payload.h")
    if(NOT EXISTS "${PAYLOAD_H}")
        message(FATAL_ERROR
            "CAELUS_EMBEDDED_UI=ON but include/ui_payload.h is missing!\n"
            "Run build.bat (Windows) or build.sh (Linux/macOS) to generate it first.")
    endif()
    target_compile_definitions(caelus_os PRIVATE CAELUS_EMBEDDED_UI=1)
    message(STATUS "UI Payload: EMBEDDED from ${PAYLOAD_H}")
else()
    message(STATUS "UI Payload: EXTERNAL (files served from ui/ directory)")
endif()

# ─── Rust Shadow-Mesh Static Library ─────────────────────────────────────────
# Accepts RUST_LIB_PATH from command-line (-DRUST_LIB_PATH=...) or auto-detects.
if(NOT RUST_LIB_PATH)
    if(WIN32)
        if(MINGW OR CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
            # MinGW/GCC needs the GNU-ABI Rust build:
            #   cargo build --release --target x86_64-pc-windows-gnu
            set(RUST_LIB_PATH
                "${CMAKE_CURRENT_SOURCE_DIR}/target/x86_64-pc-windows-gnu/release/libcaelus_network.a")
        else()
            # MSVC links the default (MSVC-ABI) build.
            set(RUST_LIB_PATH
                "${CMAKE_CURRENT_SOURCE_DIR}/target/release/caelus_network.lib")
        endif()
    else()
        set(RUST_LIB_PATH
            "${CMAKE_CURRENT_SOURCE_DIR}/target/release/libcaelus_network.a")
    endif()
endif()

if(EXISTS "${RUST_LIB_PATH}")
    target_link_libraries(caelus_os PRIVATE "${RUST_LIB_PATH}")
    message(STATUS "Rust Shadow-Mesh: ${RUST_LIB_PATH}")

    # Platform-specific system libraries required by the Rust staticlib
    if(WIN32)
        target_link_libraries(caelus_os PRIVATE
            advapi32 ws2_32 userenv bcrypt ntdll crypt32)
    elseif(APPLE)
        # macOS: Security and SystemConfiguration frameworks needed by some Rust crates
        find_library(SECURITY_LIB Security REQUIRED)
        target_link_libraries(caelus_os PRIVATE "${SECURITY_LIB}")
    else()
        # Linux
        target_link_libraries(caelus_os PRIVATE dl pthread m)
    endif()
else()
    message(WARNING
        "Rust static library not found at: ${RUST_LIB_PATH}\n"
        "Run 'cargo build --release' before CMake build.")
endif()

# ─── OR-Tools (Optional) ──────────────────────────────────────────────────────
find_package(ortools CONFIG QUIET)
if(ortools_FOUND)
    target_link_libraries(caelus_os PRIVATE ortools::ortools)
    target_compile_definitions(caelus_os PRIVATE CAELUS_WITH_ORTOOLS=1)
    message(STATUS "OR-Tools: FOUND and linked")
else()
    message(STATUS "OR-Tools: NOT found — simulation mode active")
endif()

# ─── Post-Build: Strip Debug Symbols ─────────────────────────────────────────
# Equivalent to Phase 3's 'strip --strip-all' in the build scripts.
# Provides a CMake-native mechanism for IDE builds not using build.bat/build.sh.
if(NOT MSVC AND CAELUS_STATIC_BINARY)
    find_program(STRIP_COMMAND NAMES strip llvm-strip)
    if(STRIP_COMMAND)
        add_custom_command(TARGET caelus_os POST_BUILD
            COMMAND "${STRIP_COMMAND}" --strip-all "$<TARGET_FILE:caelus_os>"
            COMMENT "[CAELUS] Stripping debug symbols from caelus_os binary..."
            VERBATIM
        )
        message(STATUS "Post-build strip: ${STRIP_COMMAND}")
    else()
        message(STATUS "Post-build strip: 'strip' not found — skipping")
    endif()
endif()

# ─── Output Location ──────────────────────────────────────────────────────────
set_target_properties(caelus_os PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/dist"
)

# ─── Summary ──────────────────────────────────────────────────────────────────
message(STATUS "")
message(STATUS "══════ CAELUS OS Build Configuration ══════")
message(STATUS "  Binary target      : dist/caelus_os")
message(STATUS "  Static binary      : ${CAELUS_STATIC_BINARY}")
message(STATUS "  Embedded UI        : ${CAELUS_EMBEDDED_UI}")
message(STATUS "  OR-Tools           : ${ortools_FOUND}")
message(STATUS "  C++ Standard       : C++${CMAKE_CXX_STANDARD}")
message(STATUS "════════════════════════════════════════════")
message(STATUS "")
